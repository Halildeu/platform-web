## ExecutiveKPIStrip

Yönetici dashboard'larında KPI metriklerini gösteren üst şerit:

```tsx
import { ExecutiveKPIStrip } from '@mfe/design-system';

<ExecutiveKPIStrip
  metrics={[
    { label: 'Toplam Gelir', value: '₺12.4M', trend: 12.5, status: 'positive' },
    { label: 'Aktif Kullanıcı', value: '8,432', trend: -2.1, status: 'negative' },
    { label: 'Tamamlanma', value: '%94', trend: 5.0, status: 'positive' },
    { label: 'SLA Uyum', value: '%99.2', trend: 0.1, status: 'neutral' },
  ]}
/>
```

```tsx
<ProcessFlow
  nodes={[

    { id: '5', type: 'task', label: 'Reddet' },
    { id: '6', type: 'end', label: 'Bitiş' },
  ]}
  edges={[
    { from: '1', to: '2' },
    { from: '2', to: '3' },
    { from: '3', to: '4', label: 'Evet' },
    { from: '3', to: '5', label: 'Hayır' },
    { from: '4', to: '6' },
    { from: '5', to: '6' },
  ]}
  onNodeClick={(node) => console.log(node)}
/>
```

## FineKinney

İş güvenliği risk değerlendirme bileşeni:

```tsx
import { FineKinney } from '@mfe/design-system';

<FineKinney
  assessment={{
    probability: 6,    // 0.1 - 10
    frequency: 3,      // 0.5 - 10
    severity: 15,      // 1 - 100
  }}
  onScoreChange={(score) => {
    // score = probability * frequency * severity
    console.log('Risk Skoru:', score);
  }}
  thresholds={{
    low: 20,
    medium: 70,
    high: 200,
    critical: 400,
  }}
/>
```

## ApprovalFlow

```tsx
<ApprovalFlow
  steps={[
    { id: '1', label: 'Yönetici Onayı', assignee: 'manager', status: 'approved' },
```
