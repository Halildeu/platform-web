import { Aside } from '@astrojs/starlight/components';

## Import

```tsx
import { DataGrid } from '@mfe/design-system';
```

## Kullanım

```tsx
const columns = [
  { field: 'id', headerName: 'ID', width: 70 },
  { field: 'name', headerName: 'Ad', flex: 1 },
  { field: 'email', headerName: 'E-posta', flex: 1 },
  { field: 'role', headerName: 'Rol', width: 120 },
];

const rows = [
  { id: 1, name: 'Ahmet Yılmaz', email: 'ahmet@ornek.com', role: 'Admin' },
  { id: 2, name: 'Ayşe Demir', email: 'ayse@ornek.com', role: 'Editör' },
];

<DataGrid columns={columns} rows={rows} />
```

## Sıralama

```tsx
<DataGrid
  columns={columns}
  rows={rows}
  sortable
  defaultSort={{ field: 'name', direction: 'asc' }}
/>
```

## Filtreleme

```tsx
<DataGrid
  columns={columns}
  rows={rows}
  filterable
  filterMode="client"  // veya "server"
/>
```

## Sayfalama

```tsx
<DataGrid
  columns={columns}
  rows={rows}
  pagination
  pageSize={25}
  pageSizeOptions={[10, 25, 50, 100]}
/>
```

## Satır Seçimi

```tsx
<DataGrid
  columns={columns}
  rows={rows}
  selectable
  selectionMode="multiple"  // veya "single"
  onSelectionChange={(selectedIds) => console.log(selectedIds)}
/>
```

## Gruplama

```tsx
<DataGrid
  columns={columns}
  rows={rows}
  groupable
  defaultGroupBy={['role']}
/>
```

## Export

```tsx
<DataGrid
  columns={columns}
  rows={rows}
  exportable
  exportFormats={['csv', 'xlsx', 'pdf']}
/>
```

## Kolon Tanımı

| Prop | Tip | Varsayılan | Açıklama |
|------|-----|-----------|----------|
| `field` | `string` | — | Veri alanı adı |
| `headerName` | `string` | — | Kolon başlığı |
| `width` | `number` | — | Sabit genişlik (px) |
| `flex` | `number` | — | Esnek genişlik oranı |
| `sortable` | `boolean` | `true` | Sıralanabilir |
| `filterable` | `boolean` | `true` | Filtrelenebilir |
| `resizable` | `boolean` | `true` | Yeniden boyutlandırılabilir |
| `renderCell` | `(params) => ReactNode` | — | Özel hücre render'ı |
| `valueFormatter` | `(value) => string` | — | Değer formatlayıcı |
| `type` | `'string' \| 'number' \| 'date' \| 'boolean'` | `'string'` | Veri tipi |

## DataGrid Props

| Prop | Tip | Varsayılan | Açıklama |
|------|-----|-----------|----------|
| `columns` | `ColumnDef[]` | — | Kolon tanımları |
| `rows` | `any[]` | — | Veri dizisi |
| `loading` | `boolean` | `false` | Yükleniyor durumu |
| `sortable` | `boolean` | `false` | Sıralama aktif |
| `filterable` | `boolean` | `false` | Filtreleme aktif |
| `pagination` | `boolean` | `false` | Sayfalama aktif |
| `pageSize` | `number` | `25` | Sayfa başına satır |
| `selectable` | `boolean` | `false` | Satır seçimi aktif |
| `selectionMode` | `'single' \| 'multiple'` | `'multiple'` | Seçim modu |
| `groupable` | `boolean` | `false` | Gruplama aktif |
| `exportable` | `boolean` | `false` | Export aktif |
| `height` | `number \| string` | `400` | Tablo yüksekliği |
| `density` | `'compact' \| 'standard' \| 'comfortable'` | `'standard'` | Satır yoğunluğu |
| `onRowClick` | `(row) => void` | — | Satır tıklama handler'ı |
| `onSelectionChange` | `(ids: any[]) => void` | — | Seçim değişikliği |

## Erişilebilirlik

- **Keyboard Navigation**: Ok tuşları ile hücre navigasyonu
- **ARIA**: `role="grid"`, `role="row"`, `role="gridcell"` otomatik
- **Focus Management**: Hücre fokus yönetimi ve tab navigasyonu
- **Screen Reader**: Kolon başlıkları ve sıralama durumu duyurulur
- **Selection**: `aria-selected` ile seçim durumu belirtilir

<Aside type="note">
  Büyük veri setleri (10,000+ satır) için sanal kaydırma (virtualization) otomatik olarak aktif olur.
</Aside>

## Server-Side İşlemler

Büyük veri setleri için sunucu taraflı sıralama, filtreleme ve sayfalama:

```tsx
<DataGrid
  columns={columns}
  rows={data}
  loading={isLoading}
  pagination
  paginationMode="server"
  rowCount={totalRows}
  page={page}
  pageSize={pageSize}
  onPageChange={setPage}
  sortingMode="server"
  onSortModelChange={handleSort}
  filterMode="server"
  onFilterModelChange={handleFilter}
/>
```
