function normalizeExpandedValue(
  value: AccordionProps["value"] | AccordionProps["defaultValue"],
): string[] {
  if (Array.isArray(value)) {
    return value.filter(
      (item): item is string => typeof item === "string" && item.length > 0,
    );
  }
  if (typeof value === "string" && value.length > 0) {
    return [value];

  return [];
}
